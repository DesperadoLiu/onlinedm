/** * 京簡康 - Google Apps Script 後端資安強化版 * 搭配前端 index.html 使用 */ // --- 安全設定區 --- const ALLOWED_ORIGIN = "https://your-username.github.io"; // 請替換為您正式上線的網址 const VALID_TOKEN = "JJK_SECURE_888"; // 務必與前端 SECURITY_TOKEN 一致 const GEMINI_API_KEY = ""; // 此處留空，執行環境會自動帶入 function doPost(e) { try { const data = JSON.parse(e.postData.contents); const userToken = data.token; const action = data.action; const userText = data.text; // 1. 安全權杖檢查 (防護風險一) if (userToken !== VALID_TOKEN) { return createResponse({ error: "未經授權的請求 (Invalid Token)" }, 403); } // 2. 基礎內容長度檢查 (防護風險二：防止惡意長文本攻擊) if (!userText || userText.length > 500) { return createResponse({ error: "請求內容無效或過長" }, 400); } // 3. 執行對應功能 if (action === "tts") { return handleTTS(userText); } else if (action === "chat") { return handleChat(userText, data.menuData || []); } return createResponse({ error: "未知的指令" }, 400); } catch (err) { return createResponse({ error: "系統錯誤: " + err.message }, 500); } } /** * 處理文字轉語音 (TTS) */ function handleTTS(text) { const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-tts:generateContent?key=${GEMINI_API_KEY}`; const payload = { contents: [{ parts: [{ text: "請用親切、活潑的口吻唸出以下介紹：" + text }] }], generationConfig: { responseModalities: ["AUDIO"], speechConfig: { voiceConfig: { prebuiltVoiceConfig: { voiceName: "Kore" } } } } }; const response = UrlFetchApp.fetch(url, { method: "post", contentType: "application/json", payload: JSON.stringify(payload) }); return createResponse(JSON.parse(response.getContentText())); } /** * 處理 AI 聊天 (Chat) */ function handleChat(prompt, menuData) { const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${GEMINI_API_KEY}`; const menuInfo = menuData.map(m => `${m.name}: ${m.desc} (價格:${m.price}元)`).join('\n'); const systemPrompt = `你現在是「京簡康」的吉祥物紫米，是一位專業的營養管家。 請根據以下菜單資訊回答客人的問題，口吻要親切、幽默且專業。 菜單內容： ${menuInfo} 如果客人問到菜單以外的問題，請禮貌地將話題引導回健康飲食與京簡康。`; const payload = { contents: [{ parts: [{ text: prompt }] }], systemInstruction: { parts: [{ text: systemPrompt }] } }; const response = UrlFetchApp.fetch(url, { method: "post", contentType: "application/json", payload: JSON.stringify(payload) }); return createResponse(JSON.parse(response.getContentText())); } /** * 建立標準回傳格式 */ function createResponse(data, statusCode = 200) { return ContentService.createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); }
